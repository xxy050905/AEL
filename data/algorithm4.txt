def select_next_node(current_node, destination_node, unvisited_nodes, distance_matrix):
    total_nodes = len(distance_matrix)
    visited_ratio = 1 - len(unvisited_nodes)/total_nodes
    if visited_ratio < 0.7:
        return max(unvisited_nodes,key=lambda x:len([n for n in unvisited_nodes if distance_matrix[x][n]<distance_matrix[current_node][x]))
    else:
        return min(unvisited_nodes, key=lambda x: distance_matrix[current_node][x])
